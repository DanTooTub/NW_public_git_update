	custom_trigger_tooltip = {
		tooltip = controls_chinese_states_tooltip
		
		all_of = {
			array = global.chinese_state
			value = state
			
			if = {
				limit = { controls_state = var:state }
				add_to_temp_variable = { num_controls_state = 1 }
			}
		}
		divide_temp_variable = { num_controls_state = global.num_chinese_states }
		log = "[THIS.GetName] [?num_controls_state|%]"
		check_variable = { num_controls_state > var_for_check }
	}
}
