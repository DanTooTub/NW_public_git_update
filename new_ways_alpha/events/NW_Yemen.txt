							has_government = monarchy
							has_government = trozkism
						}
					}
					PER = {
						OR = {
							has_government = fascism
							has_government = communism
						}
					}
					BPL = {
						OR = {
							has_government = trozkism
							has_government = communism
							has_government = fascism
						}
					}
					AND = {
						country_exists = ETH
						ETH = { NOT = { is_in_faction_with = ENG } }
					}
				}
			}
		}
		diplomatic_relation = {
			country = FROM
			relation = guarantee
			active = no
		}	
		FROM = { add_threat = 1 }
	}
	
	option = {
		name = yem.14.b
		ai_chance = {
			factor = 1
		}
	}
}
